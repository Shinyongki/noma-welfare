## 2. 데이터 구조

### 2.1 query_log 테이블 (개별 쿼리, 90일 보관)

```sql
CREATE TABLE IF NOT EXISTS query_log (
  id bigint GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  
  -- 쿼리 정보
  query_text text NOT NULL,          -- PII 마스킹된 원문
  query_length int,                   -- 원문 길이 (글자 수)
  input_method text DEFAULT 'text',   -- 'voice' | 'text'
  
  -- 매칭 결과
  matched_service text,               -- 최상위 매칭 서비스명 (null = 매칭 실패)
  top_score float,                    -- 최고 유사도 점수
  matched_faq text,                   -- 최상위 매칭 FAQ 질문 (null = 매칭 없음)
  faq_score float,                    -- FAQ 최고 유사도
  match_count int DEFAULT 0,          -- 매칭된 서비스 수
  
  -- 판정
  is_out_of_scope boolean DEFAULT false,  -- 소관 외 판정
  detected_persona text,              -- 감지된 페르소나 (elderly/proxy_child/young_parent/welfare_worker/newcomer/caregiver/general)
  detected_faq_type text,             -- 감지된 FAQ 유형 (cost/eligibility/howto/duration/content/proxy/general)
  
  -- 전환 추적
  led_to_apply boolean DEFAULT false, -- 이 검색 후 상담 신청이 이루어졌는지
  session_id text,                    -- 세션 식별 (익명, 전환 추적용)
  
  -- 메타
  created_at timestamptz DEFAULT now()
);

-- 인덱스 (집계 쿼리 성능)
CREATE INDEX IF NOT EXISTS idx_query_log_created ON query_log (created_at);
CREATE INDEX IF NOT EXISTS idx_query_log_service ON query_log (matched_service);
CREATE INDEX IF NOT EXISTS idx_query_log_scope ON query_log (is_out_of_scope);
CREATE INDEX IF NOT EXISTS idx_query_log_persona ON query_log (detected_persona);

-- RLS
ALTER TABLE query_log ENABLE ROW LEVEL SECURITY;
CREATE POLICY "query_log_insert" ON query_log FOR INSERT WITH CHECK (true);
CREATE POLICY "query_log_select" ON query_log FOR SELECT USING (true);
CREATE POLICY "query_log_delete" ON query_log FOR DELETE USING (true);
```

### 2.2 query_stats 테이블 (일별 집계, 영구 보관)

```sql
CREATE TABLE IF NOT EXISTS query_stats (
  id bigint GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  stat_date date NOT NULL UNIQUE,
  
  -- 기본 건수
  total_queries int DEFAULT 0,
  voice_queries int DEFAULT 0,
  text_queries int DEFAULT 0,
  
  -- 매칭 품질
  avg_top_score float,                -- 평균 최고 유사도
  match_failures int DEFAULT 0,       -- 매칭 실패 (top_score < 0.5) 건수
  faq_matches int DEFAULT 0,          -- FAQ 매칭 건수
  
  -- 소관 판정
  in_scope_queries int DEFAULT 0,
  out_of_scope_queries int DEFAULT 0,
  
  -- 전환
  total_applies int DEFAULT 0,        -- 상담 신청 건수
  
  -- 페르소나별 건수 (JSON)
  persona_counts jsonb DEFAULT '{}',
  -- 예: {"elderly": 30, "proxy_child": 12, "general": 10}
  
  -- FAQ 유형별 건수 (JSON)
  faq_type_counts jsonb DEFAULT '{}',
  -- 예: {"cost": 8, "eligibility": 7, "howto": 15}
  
  -- 인기 서비스 TOP 10 (JSON)
  top_services jsonb DEFAULT '[]',
  -- 예: [{"name": "긴급돌봄", "count": 15}, {"name": "노인맞춤돌봄", "count": 12}]
  
  -- 소관 외 인기 쿼리 (JSON)
  top_out_of_scope jsonb DEFAULT '[]',
  -- 예: [{"query": "기초연금 신청", "count": 5}, ...]
  
  -- 매칭 실패 인기 쿼리 (JSON)
  top_failures jsonb DEFAULT '[]',
  -- 예: [{"query": "아파요", "count": 3, "avg_score": 0.42}, ...]
  
  -- 시간대별 분포 (JSON)
  hourly_distribution jsonb DEFAULT '{}',
  -- 예: {"09": 12, "10": 15, "14": 20, ...}
  
  created_at timestamptz DEFAULT now()
);

-- 인덱스
CREATE INDEX IF NOT EXISTS idx_query_stats_date ON query_stats (stat_date);

-- RLS
ALTER TABLE query_stats ENABLE ROW LEVEL SECURITY;
CREATE POLICY "query_stats_all" ON query_stats FOR ALL USING (true);
```

### Phase B: server.js — 쿼리 로깅

#### B-1. PII 마스킹 함수

파일: `server.js`

```javascript
/**
 * 개인식별정보(PII) 마스킹
 * - 한국 전화번호: 010-1234-5678 → ***-****-****
 * - 이름 패턴: "김순자인데요" → "***인데요"
 * - 주민번호: 마스킹
 */
function maskPII(text) {
  let masked = text;
  
  // 전화번호 (다양한 형식)
  masked = masked.replace(/\d{2,4}[-.\s]?\d{3,4}[-.\s]?\d{4}/g, '***-****-****');
  
  // 주민등록번호
  masked = masked.replace(/\d{6}[-.\s]?\d{7}/g, '******-*******');
  
  // 한국 이름 패턴 (2~4글자 한글 + 조사/어미)
  // "김순자인데요", "박민수입니다", "이영희예요"
  // ⚠️ 완벽한 이름 감지는 불가능 — 합리적 수준으로만
  masked = masked.replace(
    /([가-힣]{2,4})(인데요?|입니다|이에요|예요|이요|라고\s?해요?|이라고)/g,
    '***$2'
  );
  
  return masked;
}
```

⚠️ 이름 마스킹은 완벽할 수 없음. "김해시"가 "***시"가 되지 않도록 주의.
   도시명/서비스명 화이트리스트를 추가할 수도 있지만, 초기에는 단순 패턴으로 시작.

#### B-2. 페르소나 감지 함수

```javascript
/**
 * 발화 패턴으로 사용자 페르소나 추정
 */
function detectPersona(queryText, isVoiceInput) {
  const q = queryText.toLowerCase();
  
  // 음성 + 짧은 문장 → elderly
  if (isVoiceInput && queryText.length < 20) return 'elderly';
  
  // 가족 대리 키워드
  if (/어머니|아버지|부모님|할머니|할아버지|엄마|아빠/.test(q)) return 'proxy_child';
  
  // 젊은 부모 키워드
  if (/장애.*아동|아이.*장애|어린이집|보조기기.*아이|발달/.test(q)) return 'young_parent';
  
  // 복지 실무자 키워드
  if (/대상자|이관|연계.*절차|법.*시행|대상.*차이|근거/.test(q)) return 'welfare_worker';
  
  // 전입 주민 키워드
  if (/이사|전입|처음|경남.*서비스|뭐가\s?있/.test(q)) return 'newcomer';
  
  // 종사자 키워드
  if (/활동지원사|대체인력|컨설팅|교육.*일정|종사자/.test(q)) return 'caregiver';
  
  // 짧고 감정적 → elderly
  if (queryText.length < 15 && /아파|무서|힘들|외로|배고|도와|사람/.test(q)) return 'elderly';
  
  return 'general';
}
```

### Phase C: server.js — 일별 집계

#### C-1. 집계 함수

```javascript
/**
 * 어제 날짜의 query_log를 집계하여 query_stats에 저장
 * 서버 시작 시 + 매일 자정(KST) 자동 실행
 */
async function aggregateDailyStats() {
  if (!supabase) return;
  
  try {
    const yesterday = new Date();
    yesterday.setDate(yesterday.getDate() - 1);
    const dateStr = yesterday.toISOString().split('T')[0];  // "YYYY-MM-DD"
    
    const startOfDay = `${dateStr}T00:00:00+09:00`;  // KST
    const endOfDay = `${dateStr}T23:59:59+09:00`;
    
    // 해당 날짜 로그 조회
    const { data: logs, error } = await supabase
      .from('query_log')
      .select('*')
      .gte('created_at', startOfDay)
      .lte('created_at', endOfDay);
    
    if (error || !logs || logs.length === 0) return;
    
    // 집계 계산
    const stats = {
      stat_date: dateStr,
      total_queries: logs.length,
      voice_queries: logs.filter(l => l.input_method === 'voice').length,
      text_queries: logs.filter(l => l.input_method === 'text').length,
      avg_top_score: logs.reduce((s, l) => s + (l.top_score || 0), 0) / logs.length,
      match_failures: logs.filter(l => !l.top_score || l.top_score < 0.5).length,
      faq_matches: logs.filter(l => l.faq_score && l.faq_score > 0.5).length,
      in_scope_queries: logs.filter(l => !l.is_out_of_scope).length,
      out_of_scope_queries: logs.filter(l => l.is_out_of_scope).length,
      total_applies: logs.filter(l => l.led_to_apply).length,
      persona_counts: countBy(logs, 'detected_persona'),
      faq_type_counts: countBy(logs, 'detected_faq_type'),
      top_services: topN(logs.filter(l => l.matched_service), 'matched_service', 10),
      top_out_of_scope: topNQueries(logs.filter(l => l.is_out_of_scope), 10),
      top_failures: topNFailures(logs.filter(l => !l.top_score || l.top_score < 0.5), 10),
      hourly_distribution: hourlyCount(logs),
    };
    
    // upsert (날짜 중복 시 덮어쓰기)
    await supabase.from('query_stats').upsert(stats, { onConflict: 'stat_date' });
    
    console.log(`[집계] ${dateStr} 완료: ${logs.length}건`);
  } catch (err) {
    console.error('[집계 오류]', err.message);
  }
}

// 헬퍼 함수들
function countBy(arr, field) {
  return arr.reduce((acc, item) => {
    const key = item[field] || 'unknown';
    acc[key] = (acc[key] || 0) + 1;
    return acc;
  }, {});
}

function topN(arr, field, n) {
  const counts = countBy(arr, field);
  return Object.entries(counts)
    .map(([name, count]) => ({ name, count }))
    .sort((a, b) => b.count - a.count)
    .slice(0, n);
}

function topNQueries(arr, n) {
  const counts = {};
  arr.forEach(l => {
    const q = l.query_text?.substring(0, 50) || 'unknown';
    counts[q] = (counts[q] || 0) + 1;
  });
  return Object.entries(counts)
    .map(([query, count]) => ({ query, count }))
    .sort((a, b) => b.count - a.count)
    .slice(0, n);
}

function topNFailures(arr, n) {
  const grouped = {};
  arr.forEach(l => {
    const q = l.query_text?.substring(0, 50) || 'unknown';
    if (!grouped[q]) grouped[q] = { query: q, count: 0, totalScore: 0 };
    grouped[q].count++;
    grouped[q].totalScore += (l.top_score || 0);
  });
  return Object.values(grouped)
    .map(g => ({ query: g.query, count: g.count, avg_score: +(g.totalScore / g.count).toFixed(3) }))
    .sort((a, b) => b.count - a.count)
    .slice(0, n);
}

function hourlyCount(arr) {
  const hours = {};
  arr.forEach(l => {
    const h = new Date(l.created_at).getHours().toString().padStart(2, '0');
    hours[h] = (hours[h] || 0) + 1;
  });
  return hours;
}
```

#### C-2. 자동 실행 스케줄

```javascript
// 서버 시작 시 어제 집계 실행 (누락 방지)
aggregateDailyStats();

// 매일 자정(KST) 집계 — setInterval 대신 시간 체크
setInterval(() => {
  const now = new Date();
  const kstHour = (now.getUTCHours() + 9) % 24;
  const kstMinute = now.getUTCMinutes();
  // KST 00:05 (자정 5분 후)에 집계
  if (kstHour === 0 && kstMinute >= 5 && kstMinute < 6) {
    aggregateDailyStats();
  }
}, 60000); // 1분마다 체크
```

#### C-3. 90일 자동 정리

```javascript
/**
 * 90일 이상 된 query_log 삭제
 * 집계 후 실행 (query_stats에 이미 집계되었으므로 안전)
 */
async function cleanOldLogs() {
  if (!supabase) return;
  
  try {
    const cutoff = new Date();
    cutoff.setDate(cutoff.getDate() - 90);
    
    const { count, error } = await supabase
      .from('query_log')
      .delete()
      .lt('created_at', cutoff.toISOString());
    
    if (!error) {
      console.log(`[정리] 90일 이전 로그 삭제: ${count || 0}건`);
    }
  } catch (err) {
    console.error('[로그 정리 오류]', err.message);
  }
}

// 집계 후 정리 실행
async function dailyMaintenance() {
  await aggregateDailyStats();
  await cleanOldLogs();
}
```

### Phase D: server.js — 분석 API 엔드포인트

#### D-1. 분석 데이터 조회 API

```javascript
// 질의 분석 — 기간별 집계 데이터
// GET /api/admin/query-analytics?period=7|14|30|90
app.get('/api/admin/query-analytics', requireAuth, async (req, res) => {
  try {
    const period = parseInt(req.query.period) || 14;
    const since = new Date();
    since.setDate(since.getDate() - period);
    const sinceStr = since.toISOString().split('T')[0];
    
    // query_stats에서 기간 데이터 조회
    const { data: stats, error } = await supabase
      .from('query_stats')
      .select('*')
      .gte('stat_date', sinceStr)
      .order('stat_date', { ascending: true });
    
    if (error) throw error;
    
    // 기간 합산 집계
    const summary = {
      period,
      totalQueries: 0,
      voiceQueries: 0,
      textQueries: 0,
      avgTopScore: 0,
      matchFailures: 0,
      faqMatches: 0,
      inScope: 0,
      outOfScope: 0,
      totalApplies: 0,
      conversionRate: 0,       // 검색→상담신청 전환율
      outOfScopeRate: 0,       // 소관 외 비율
      voiceRate: 0,            // 음성 입력 비율
      failureRate: 0,          // 매칭 실패율
    };
    
    const personaCounts = {};
    const faqTypeCounts = {};
    const serviceCounts = {};
    const outOfScopeCounts = {};
    const failureCounts = {};
    const hourlyTotal = {};
    
    stats.forEach(s => {
      summary.totalQueries += s.total_queries;
      summary.voiceQueries += s.voice_queries;
      summary.textQueries += s.text_queries;
      summary.matchFailures += s.match_failures;
      summary.faqMatches += s.faq_matches;
      summary.inScope += s.in_scope_queries;
      summary.outOfScope += s.out_of_scope_queries;
      summary.totalApplies += s.total_applies;
      
      // 페르소나 합산
      if (s.persona_counts) {
        Object.entries(s.persona_counts).forEach(([k, v]) => {
          personaCounts[k] = (personaCounts[k] || 0) + v;
        });
      }
      
      // FAQ 유형 합산
      if (s.faq_type_counts) {
        Object.entries(s.faq_type_counts).forEach(([k, v]) => {
          faqTypeCounts[k] = (faqTypeCounts[k] || 0) + v;
        });
      }
      
      // 서비스 합산
      if (s.top_services) {
        s.top_services.forEach(svc => {
          serviceCounts[svc.name] = (serviceCounts[svc.name] || 0) + svc.count;
        });
      }
      
      // 시간대 합산
      if (s.hourly_distribution) {
        Object.entries(s.hourly_distribution).forEach(([h, c]) => {
          hourlyTotal[h] = (hourlyTotal[h] || 0) + c;
        });
      }
    });
    
    // 비율 계산
    if (summary.totalQueries > 0) {
      summary.avgTopScore = stats.reduce((s, d) => s + (d.avg_top_score || 0), 0) / stats.length;
      summary.conversionRate = +(summary.totalApplies / summary.totalQueries * 100).toFixed(1);
      summary.outOfScopeRate = +(summary.outOfScope / summary.totalQueries * 100).toFixed(1);
      summary.voiceRate = +(summary.voiceQueries / summary.totalQueries * 100).toFixed(1);
      summary.failureRate = +(summary.matchFailures / summary.totalQueries * 100).toFixed(1);
    }
    
    res.json({
      summary,
      daily: stats,  // 일별 추이 차트용
      personaCounts,
      faqTypeCounts,
      topServices: Object.entries(serviceCounts)
        .map(([name, count]) => ({ name, count }))
        .sort((a, b) => b.count - a.count)
        .slice(0, 10),
      hourlyDistribution: hourlyTotal,
    });
  } catch (err) {
    console.error('[분석 API 오류]', err.message);
    res.status(500).json({ error: '분석 데이터 조회 실패' });
  }
});

// 매칭 실패 쿼리 목록 (최근 90일, query_log에서 직접 조회)
// GET /api/admin/query-failures?limit=20
app.get('/api/admin/query-failures', requireAuth, async (req, res) => {
  try {
    const limit = Math.min(parseInt(req.query.limit) || 20, 100);
    
    const { data, error } = await supabase
      .from('query_log')
      .select('query_text, top_score, matched_service, input_method, detected_persona, created_at')
      .or('top_score.is.null,top_score.lt.0.5')
      .order('created_at', { ascending: false })
      .limit(limit);
    
    if (error) throw error;
    res.json(data || []);
  } catch (err) {
    res.status(500).json({ error: '실패 쿼리 조회 실패' });
  }
});

// 소관 외 쿼리 목록 (최근 90일)
// GET /api/admin/query-out-of-scope?limit=20
app.get('/api/admin/query-out-of-scope', requireAuth, async (req, res) => {
  try {
    const limit = Math.min(parseInt(req.query.limit) || 20, 100);
    
    const { data, error } = await supabase
      .from('query_log')
      .select('query_text, top_score, input_method, detected_persona, created_at')
      .eq('is_out_of_scope', true)
      .order('created_at', { ascending: false })
      .limit(limit);
    
    if (error) throw error;
    res.json(data || []);
  } catch (err) {
    res.status(500).json({ error: '소관 외 쿼리 조회 실패' });
  }
});
```

⚠️ 모든 분석 API에 `requireAuth` 미들웨어 적용 필수.

## 5. 주의사항

### 절대 변경 금지
- 기존 analyticsStore 로직 (기존 일별 건수 기록은 유지)
- 기존 분석 탭의 차트 (서비스 인기도, 카테고리 분포, KPI 추이, 이벤트 추이)
- withLock() 패턴
- /api/chat 응답 성능 (로깅은 비동기)

### 성능 주의
- query_log INSERT는 비동기로 처리 — 응답 지연 없도록
- 분석 API는 query_stats (집계 데이터) 기반 — 큰 쿼리 안 돌림
- 실패/소관외 목록만 query_log 직접 조회 (최근 N건 제한)

### Supabase 제약
- Phase A SQL은 사용자에게 SQL Editor 실행 요청
- anon key로 INSERT/SELECT/DELETE 가능 (RLS 정책 포함)
- supabase 미연결 시 로깅 자체를 스킵 (에러 없이)

### 개인정보 주의
- query_text는 반드시 maskPII() 후 저장
- 이름 마스킹이 불완전할 수 있음 — 로그 열람 권한은 관리자만
- query_log 90일 자동 정리로 장기 보관 리스크 최소화

### 롤백 계획
- 로깅 코드: INSERT 호출 부분만 주석 처리하면 원복
- 집계 코드: setInterval/aggregateDailyStats 주석 처리
- UI: 질의 분석 섹션을 display:none으로 숨김
