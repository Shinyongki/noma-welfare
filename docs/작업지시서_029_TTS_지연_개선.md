# 작업지시서_029_TTS_지연_개선
# 작성일: 2026-03-15
# 목적: 음성 응답 지연 단축 — 정형 질문 사전 캐시 + 첫 문장 조기 TTS

---

## 현재 문제

사용자 입력 후 음성이 시작되기까지 **3~8초** 소요.

| 구간 | 소요 시간 | 병목 여부 |
|------|---------|---------|
| Gemini 스트리밍 | 2~5초 | ★ 주요 병목 |
| Edge TTS 합성 | 1~3초 | ★ 주요 병목 |
| 디코딩 + 필터 | ~200ms | 부차적 |

현재는 Gemini 완료 → TTS 호출 → 합성 완료 → 재생 순서로 직렬 처리됨.

---

## 개선 목표

**개선 1. 정형 질문 사전 캐시** → Edge TTS 합성 시간 0ms로 단축
**개선 2. 첫 문장 조기 TTS** → Gemini 스트리밍 중 음성 시작

두 개선을 합치면 체감 지연이 **3~8초 → 1~2초** 수준으로 단축됨.

---

## 개선 1: 정형 질문 사전 캐시

### 1-1. server.js — 서버 시작 시 TTS 캐시 생성

서버 시작 시 아래 고정 문구 5개를 미리 TTS 합성하여 메모리에 저장한다.

**추가 위치**: `synthesizeTTS` 함수 선언 직후 (server.js:2817 이후)

```javascript
// ── TTS 사전 캐시 ──────────────────────────────────────────
const TTS_CACHE = new Map();

const TTS_PRESET_PHRASES = [
  "상담 신청을 위해 성함을 알려주시겠어요?",
  "연락받으실 전화번호도 알려주시겠어요?",
  "네, 알겠습니다. 잠시만 기다려 주세요.",
  "감사합니다. 담당 기관에 상담 요청을 보내드렸습니다.",
  "죄송합니다. 다시 한번 말씀해 주시겠어요?",
];

async function warmupTTSCache() {
  console.log('[TTS] 사전 캐시 생성 시작...');
  for (const phrase of TTS_PRESET_PHRASES) {
    try {
      const audio = await synthesizeTTS(phrase);
      if (audio) {
        TTS_CACHE.set(phrase.trim(), audio);
        console.log(`[TTS] 캐시 완료: "${phrase.slice(0, 20)}..."`);
      }
    } catch (e) {
      console.warn(`[TTS] 캐시 실패: ${e.message}`);
    }
  }
  console.log(`[TTS] 사전 캐시 완료 (${TTS_CACHE.size}건)`);
}

// 서버 시작 후 비동기로 워밍업 (서버 시작을 블로킹하지 않음)
setTimeout(warmupTTSCache, 3000);
```

### 1-2. server.js — /api/tts 엔드포인트 캐시 우선 반환

**위치**: `/api/tts` 엔드포인트 내부, `analyticsStore.track('tts_request')` 직후

```javascript
// 캐시 히트 시 즉시 반환
const cacheKey = String(text).trim();
if (TTS_CACHE.has(cacheKey)) {
  console.log('[TTS] 캐시 히트:', cacheKey.slice(0, 20));
  return res.json({ audioContent: TTS_CACHE.get(cacheKey), cached: true });
}
```

## 개선 2: 첫 문장 조기 TTS

Gemini 스트리밍 중 첫 문장이 완성되는 시점에 바로 TTS를 호출한다.
첫 문장 재생이 끝나면 나머지 텍스트를 이어서 재생한다.

### 2-1. code.html — SSE 스트림 처리 중 첫 문장 감지 및 조기 호출

**위치**: SSE 스트림 처리 블록에서 `type: 'stream'` 이벤트 처리 부분
(현재 Gemini 청크를 받아 화면에 렌더링하는 로직 내부)

아래 변수를 검색 시작 시 초기화하는 부분에 추가할 것:
```javascript
let firstTTSSent = false;         // 첫 문장 TTS 전송 여부
let streamAccumulated = '';       // 스트리밍 누적 텍스트 (첫 문장 감지용)
let firstTTSPromise = null;       // 첫 문장 TTS Promise
```

SSE `type: 'stream'` 이벤트를 받는 부분에 아래 로직을 추가할 것:
(기존 화면 렌더링 코드를 건드리지 말고, 그 직후에 추가)

```javascript
// ── 첫 문장 조기 TTS ──────────────────────────────────────
if (ttsEnabled && !firstTTSSent) {
  streamAccumulated += data.text;

  // 문장 종결 감지: 마침표/물음표/느낌표 + 공백 또는 줄바꿈
  const sentenceEndMatch = streamAccumulated.match(/^(.{20,}[.?!])\s/);
  if (sentenceEndMatch) {
    firstTTSSent = true;
    const firstSentence = sentenceEndMatch[1];
    // noma-card, noma-apply 태그 포함 시 조기 TTS 스킵
    if (!firstSentence.includes('<noma-card>') && !firstSentence.includes('<noma-apply>')) {
      console.log('[TTS] 첫 문장 조기 호출:', firstSentence.slice(0, 30));
      firstTTSPromise = speakText(firstSentence);
    }
  }
}
```

### 2-2. code.html — 스트리밍 완료 후 나머지 텍스트 처리

**위치**: 기존 TTS 호출 부분 (현재 `speakText(extractSpeakableText(fullResponse))` 호출 부분)

```javascript
// 기존 코드 교체:
// const ttsPromise = (fullResponse && ttsEnabled)
//     ? speakText(extractSpeakableText(fullResponse))
//     : Promise.resolve();

// 변경 코드:
let ttsPromise = Promise.resolve();
if (fullResponse && ttsEnabled) {
  if (firstTTSSent && firstTTSPromise) {
    // 첫 문장은 이미 재생 중 — 나머지 텍스트 추출 후 이어서 재생
    const speakable = extractSpeakableText(fullResponse);
    const firstSentenceEnd = speakable.match(/^(.{20,}[.?!])\s/);
    if (firstSentenceEnd) {
      const remainder = speakable.slice(firstSentenceEnd[1].length).trim();
      // 첫 문장 재생 완료 후 나머지 재생
      ttsPromise = firstTTSPromise.then(() => {
        if (remainder.length > 10) return speakText(remainder);
      });
    } else {
      ttsPromise = firstTTSPromise;
    }
  } else {
    // 첫 문장 감지 실패 시 기존 방식 유지
    ttsPromise = speakText(extractSpeakableText(fullResponse));
  }
}
```

## 주의 사항

- `warmupTTSCache()`는 3초 딜레이 후 실행되므로 서버 시작 직후에는 캐시 미적용. 정상 동작임.
- 첫 문장 감지 최소 길이는 20자로 설정. 너무 짧은 문장(인사말 등)에 TTS가 먼저 트리거되는 것을 방지.
- `noma-card`, `noma-apply` 태그가 포함된 스트리밍 청크에서는 조기 TTS를 스킵. 카드 JSON이 음성으로 읽히는 것을 방지.
- 기존 `stopSpeaking()` 호출 로직(새 검색 시 TTS 중단)은 그대로 유지할 것.
- Railway 환경에서 Edge TTS 외부 API 응답 시간이 로컬과 다를 수 있음. 배포 후 실제 체감 확인 필요.
