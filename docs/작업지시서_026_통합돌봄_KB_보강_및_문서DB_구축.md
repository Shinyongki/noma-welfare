# 작성일: 2026-03-07

## 1. 목표

2026년 3월 27일 돌봄통합지원법 전국 시행에 대비하여:
- **A**: 도민용 핵심 정보를 welfare_kb + welfare_faq에 구조화하여 추가
- **B**: 상세 실무 문서를 새 welfare_docs 테이블에 청크 단위로 저장
- **C**: server.js에 3개 테이블 병렬 검색 통합
- **D**: 시스템 프롬프트에 통합돌봄 질의 대응 가이드 추가

**핵심 원칙**: 통합돌봄 신청 창구는 **읍면동 행정복지센터 또는 건보공단**이며, 경상남도사회서비스원이 아님. 이 구분이 모든 안내의 출발점.

## 5. Phase C: welfare_docs 테이블 구축 (상세 문서 DB)

### C-1. Supabase 테이블 생성 (SQL Editor 수동 실행)

```sql
-- 문서 청크 저장 테이블
CREATE TABLE IF NOT EXISTS welfare_docs (
  id BIGSERIAL PRIMARY KEY,
  
  -- 문서 메타데이터
  doc_source TEXT NOT NULL,           -- 문서 출처 ('통합돌봄_표준교안_읍면동_260226')
  doc_title TEXT NOT NULL,            -- 문서 제목
  doc_type TEXT NOT NULL,             -- 'guideline', 'manual', 'policy', 'case_study'
  
  -- 청크 정보
  section TEXT NOT NULL,              -- 섹션명 (예: 'IV.업무절차-01.통합지원업무절차')
  page_numbers TEXT,                  -- 원본 페이지 (예: '17-18')
  chunk_index INTEGER NOT NULL,       -- 같은 섹션 내 순서
  
  -- 내용
  content TEXT NOT NULL,              -- 청크 텍스트
  summary TEXT,                       -- 청크 요약 (검색 결과 표시용)
  
  -- 대상 사용자
  target_audience TEXT DEFAULT 'staff', -- 'citizen', 'staff', 'both'
  
  -- 임베딩
  embedding vector(3072),             -- gemini-embedding-001
  
  -- 메타
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 검색용 인덱스 (3072차원 > 2000 제한으로 벡터 인덱스 불가, 메타데이터 인덱스만)
CREATE INDEX idx_welfare_docs_source ON welfare_docs(doc_source);
CREATE INDEX idx_welfare_docs_type ON welfare_docs(doc_type);
CREATE INDEX idx_welfare_docs_audience ON welfare_docs(target_audience);

-- 벡터 검색 함수
CREATE OR REPLACE FUNCTION match_welfare_docs(
  query_embedding vector(3072),
  match_threshold float DEFAULT 0.65,
  match_count int DEFAULT 5,
  audience_filter text DEFAULT NULL
)
RETURNS TABLE (
  id bigint,
  doc_source text,
  doc_title text,
  section text,
  page_numbers text,
  content text,
  summary text,
  target_audience text,
  similarity float
)
LANGUAGE plpgsql
AS $$
BEGIN
  RETURN QUERY
  SELECT
    wd.id,
    wd.doc_source,
    wd.doc_title,
    wd.section,
    wd.page_numbers,
    wd.content,
    wd.summary,
    wd.target_audience,
    1 - (wd.embedding <=> query_embedding) AS similarity
  FROM welfare_docs wd
  WHERE 1 - (wd.embedding <=> query_embedding) > match_threshold
    AND (audience_filter IS NULL OR wd.target_audience = audience_filter OR wd.target_audience = 'both')
  ORDER BY wd.embedding <=> query_embedding
  LIMIT match_count;
END;
$$;
```

## 6. Phase D: server.js 3테이블 병렬 검색 통합

### D-1. welfare_docs 검색 함수 추가

```javascript
/**
 * welfare_docs 벡터 검색
 * @param {Array} queryEmbedding - 쿼리 임베딩 (3072차원)
 * @param {number} threshold - 유사도 임계값
 * @param {number} count - 반환 건수
 * @param {string} audience - 대상 ('citizen', 'staff', null=전체)
 */
async function pgvectorDocSearch(queryEmbedding, threshold = 0.65, count = 3, audience = null) {
  if (!supabase) return [];
  
  try {
    const { data, error } = await supabase.rpc('match_welfare_docs', {
      query_embedding: queryEmbedding,
      match_threshold: threshold,
      match_count: count,
      audience_filter: audience,
    });
    
    if (error) {
      console.error('[DocSearch] 오류:', error.message);
      return [];
    }
    
    return data || [];
  } catch (err) {
    console.error('[DocSearch] 예외:', err.message);
    return [];
  }
}
```
